## 풀이
---
- 문제에서 어떻게 해라라고 조건이 있어 그대로 따라서 코드를 작성하면 되는 문제였다.
```jsx
function solution(n) {
    let result = 0
    let hex = n.toString(2)
    let count = 0
    for(let el of hex){
        if(el === '1') count++
    }
    
    let flag=1
    while(flag){
      n = n+1  
      let hex = n.toString(2)
      let findCount= 0
      for(let el of hex){
         if(el === '1') findCount++
      }
       if(count === findCount) {
           result = n
           flag=0
       }
    }

    return result
}

```
