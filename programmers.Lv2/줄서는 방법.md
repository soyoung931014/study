function solution(n, k) {
  let ch = Array.from({length: n+1}, () => 0)
  let answer = []
  let tmp=[]
  
  function DFS(L){
      if(L===n){
          answer.push(tmp.slice())
      }
      else{
          for(let i=1; i<=n; i++){
              if(ch[i] === 0){
                  ch[i]=1
                  tmp.push(i)
                  DFS(L+1)
                  tmp.pop()
                  ch[i]=0
              }
          }
      }
  }
  DFS(0)
  return answer[k-1]  
}
