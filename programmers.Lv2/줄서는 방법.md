## 풀이
----
- 간단하게 순열로 접근하면 된다고 생각했는데, 효율성 측면에서 빵점을 받았다.
```jsx
function solution(n, k) {
  let ch = Array.from({length: n+1}, () => 0)
  let answer = []
  let tmp=[]
  
  function DFS(L){
      if(L===n){
          answer.push(tmp.slice())
      }
      else{
          for(let i=1; i<=n; i++){
              if(ch[i] === 0){
                  ch[i]=1
                  tmp.push(i)
                  DFS(L+1)
                  tmp.pop()
                  ch[i]=0
              }
          }
      }
  }
  DFS(0)
  return answer[k-1]  
}

```
