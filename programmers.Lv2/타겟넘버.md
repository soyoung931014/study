function solution(numbers, target) {
    let count = 0
    let n = numbers.length
    function DFS(L, sum){
        if(L === n) {
            if(sum === target) {
                count++
            }
            return
        }
        else{
            DFS(L+1, sum+numbers[L])
            DFS(L+1, sum-numbers[L])
        }
    }
    DFS(0,0)
    return count
}
