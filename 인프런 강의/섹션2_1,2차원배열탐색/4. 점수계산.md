            function solution(arr){         
              let result = 0
              let count = 0
              for(let el of arr) {
                if(el === 1) {
                    count++ 
                    result += count 
                }else{
                    count = 0
                }
              }
              return result
            }

            let arr=[1, 0, 1, 1, 1, 0, 0, 1, 1, 0];
            console.log(solution(arr)); // 10
        </script>
