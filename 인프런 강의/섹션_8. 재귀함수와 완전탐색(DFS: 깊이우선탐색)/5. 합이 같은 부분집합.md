            function solution(arr){
                let answer = "NO"
                let total = arr.reduce((acc,cur) => acc+cur, 0)
                let n=arr.length; // 6
                function DFS(L, sum){
                    if(L === n) {
                        //부분집합완성
                        if((total - sum) === sum) answer="YES"
                    }else{
                        DFS(L+1, sum+arr[L]) // 사용하면
                        DFS(L+1, sum) // 사용안하면
                    }
                }
                DFS(0,0)
                return answer;
            }
 
            let arr=[1, 3, 5, 6, 7, 10]; // yes
            console.log(solution(arr));
        </script>
