        <script>
            function solution(s){  
              let answer;
              stack=[]
              for(let x of s) {
                if(x === ')') {
                    while (stack.pop() !== "(");
                }
                else stack.push(x)
              }
              answer=stack.join("")
              return answer
            }
            

           let str = "ER(ERER)"
            console.log(solution(str));
        </script>
