```jsx
<html>
    <head>
        <meta charset="UTF-8">
        <title>출력결과</title>
    </head>
    <body>
        <script>
            function solution(meeting){
                let startTime = 0
                let count = 0
                let answer = meeting
                meeting.sort((a,b) => {
                    if(a[1] === b[1])  return a[0]-b[0] // 끝나는 시간이 같으면 시작 시간이 같다.
                    else return a[1] - b[1]
                }) // 기본적으로 [1]번째로 sort를 하는데 a[1] === b[1]이 같은 경우에는 [0]번째로 순서 정한다.
                for(let x of meeting) {
                    if(x[0] >= startTime) {
                        count++
                        startTime = x[1]
                    }
                }
                return count
            }

            let arr=[[1, 4], [2, 3], [3, 5], [4, 6], [5, 7]];
            console.log(solution(arr));
        </script>
    </body>
</html>
```


### 또품
```jsx
<html>
    <head>
        <meta charset="UTF-8">
        <title>출력결과</title>
    </head>
    <body>
        <script>
            function solution(meeting){
              let cnt=0
              let et = 0
              meeting.sort((a,b) => {
                if(a[1] === b[1]) return a[0]-b[0]
                else return a[1]-b[1]
              })
              for(let [s,e] of meeting){
                if(et <= s) {
                    et= e
                    cnt++
                }
              }
              return cnt
            }

            let arr=[[1, 4], [2, 3], [3, 5], [4, 6], [5, 7]];
            console.log(solution(arr));
        </script>
    </body>
</html>
```
