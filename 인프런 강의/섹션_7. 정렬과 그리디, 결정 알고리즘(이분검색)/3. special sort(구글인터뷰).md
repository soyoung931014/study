// 버블 정렬로 풀어본것 => 참고로 버블정렬은 성능이 좋진 못함
<html>
    <head>
        <meta charset="UTF-8">
        <title>출력결과</title>
    </head>
    <body>
        <script>
            function solution(arr){
               for(let i =0; i<arr.length-1; i++) {
                    for(let j = 0; j< arr.length-i-1; j++) {
                        if(arr[j] > arr[j+1]) {
                            [arr[j], arr[j+1]] = [arr[j+1],arr[j]]
                        }
                    }
               }
               return arr
            }

            let arr=[1, 2, 3, -3, -2, 5, 6, -6];
            console.log(solution(arr));
        </script>
    </body>
</html>
