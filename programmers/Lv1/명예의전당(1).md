## 풀이

```jsx

// 맨 밑 풀이법 코드 정리
function solution(k, score) {
    let result =[]
    let tmp=[]
    for(let i=0; i<score.length; i++){
            tmp.push(score[i])
            tmp.sort((a,b) => b-a)
            if(tmp.length > k) tmp.pop()
            result.push(tmp[tmp.length-1])
    }
    return result
}



// 반복문을 돌려 tmp.length의 기준에 따라서 분기해준다.
// tmp.length < k 일경우 tmp의 최솟값을 result배열에 푸시한다
// tmp === k일경우는 내림차순으로 sort한 후, 마지막요소를 result배열에 푸시한다.
// tmp > k일 경우는 내림차순으로 sort한 후, 마지막 요소 pop(불필요한 요소)해주고(왜냐면 우리가 구하는것은 k 등수가 필요하기때문 2번째 예시충족) 마지막 요소를 result배열에 푸시한다.

function solution(k, score) {
    let result =[]
    let tmp=[]
    for(let i=0; i<score.length; i++){
            tmp.push(score[i])
            if(tmp.length < k){
                let element = Math.min(...tmp)
                result.push(element)
            }else if(tmp.length === k){
                tmp.sort((a,b) => b-a)
                result.push(tmp[tmp.length-1])
            }else{
                tmp.sort((a,b) => b-a)
                tmp.pop()
                result.push(tmp[tmp.length-1])
            }
            
    }
    return result
}
```

function solution(k, score) {
   let result =[]
   let tmp=[]
   for(let i=0; i<score.length; i++){
       if(i<k){
           tmp.push(score[i])
           tmp.sort((a,b) => b-a)
           result.push(tmp[tmp.length-1])
       }else{
           if(tmp[tmp.length-1] < score[i]){
               tmp.pop()
               tmp.push(score[i])
               tmp.sort((a,b) => b-a)
           }
           result.push(tmp[tmp.length-1])
       }
   } 
    return result
}
