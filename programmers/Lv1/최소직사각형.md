```jsx
function solution(sizes) {
    const rotated = sizes.map(([w,h]) => w < h ? [h,w]:[w,h])
    let maxSize=[0,0]
    rotated.forEach(([w,h]) => {
        if(w>maxSize[0]) maxSize[0] = w
        if(h>maxSize[1]) maxSize[1] = h
    })
    return maxSize[0]*maxSize[1]
}
```

### 또풀이
```jsx
function solution(sizes){
    let first=[]
    let second=[]
    for(let i=0; i<sizes.length; i++){
        let fir = sizes[i][0]
        let sec = sizes[i][1]
        
        if(fir<sec){
            fir=sizes[i][1]
            sec=sizes[i][0]
        }
        first.push(fir)
        second.push(sec)
    }
    return Math.max(...first) * Math.max(...second)
    
}

```

```jsx
function solution(sizes) {
   let arr = sizes
   arr = arr.map(el => el.sort((a,b) => b-a))
   let firMax=0
   let secMax=0 
   for(let [first,sec] of arr){
       if(first>firMax) firMax=first
       if(sec>secMax) secMax=sec
   } 
    return firMax*secMax
}
```

### 또품
```jsx
function solution(sizes) {
    let firMax=Number.MIN_SAFE_INTEGER
    let secMax=Number.MIN_SAFE_INTEGER
    for(let arr of sizes){
        if(arr[0] > arr[1]) arr=[arr[1], arr[0]]
        firMax = Math.max(arr[0], firMax)
        secMax = Math.max(arr[1], secMax)
    }
    return firMax * secMax
}
```
