### 풀이
----
```jsx
function solution(price, money, count) {
    let i = 1
    let answer = 0
    while(i <= count) {
        answer += price * i
        i++
    }
    return answer-money >=0 ? answer-money : 0  
}

```

```jsx
function solution(price, money, count) {
    let result = 0
    for(let i =1; i<=count; i++) {
        result = result + price*i
    }
    
    if(money > result) {
        return 0
    }
    return result - money  
}
```

}
