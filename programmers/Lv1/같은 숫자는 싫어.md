function solution(arr){
    let result = []
    for(let i = 0; i<arr.length; i++) {
        if(arr[i] !== arr[i+1]) result.push(arr[i])
    }
    return result
    
}

### 풀이

```jsx
function solution(arr){
    let result = []
    for(let i=0; i<arr.length; i++){
        let j
        for(j=i+1; j<arr.length; j++){
            if(arr[i] === arr[j]) continue
            else {
                result.push(arr[i])
                i=j-1
                break
            }
        }
    }
    result.push(arr[arr.length-1])
    return result
}
```
