function solution(s) {
    let stack = []
    let cnt = 0
   for(let i=0; i<s.length; i++){
       stack.push(s[i])
       let storeX = stack.filter(item => item === stack[0])
       let notStoreX = stack.filter(item => item !== stack[0])
  // ğŸ“‹ console.log(storeX, notStoreX)
       if(storeX.length === notStoreX.length){
           cnt++
           stack=[]
       }
   }
    if(stack.length > 0) cnt++
    return cnt
}
