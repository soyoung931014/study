# 풀이
```jsx
function solution(a, b) {
    let result = 0
    for(let i =0; i<a.length; i++) {
        result= result + a[i]*b[i]
    }
    return result
   
}
```
```jsx
function solution(a, b) {

      let result = a.reduce((acc, cur, idx) => acc += cur*b[idx],0) ✅ reduce에서 초기값 설정해줘야, idx가 0부터 시작한다., 설정안해주면 1부터 시작됨
      return result
}
```
