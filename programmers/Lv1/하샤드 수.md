### 풀이
```jsx
function solution(x) {

    let argument = String(x).split('')   
    let result = 0
    
    for(let i = 0; i<argument.length; i++) {
      result = result + Number(argument[i])
    }
   
    if(x%result === 0) {
      return true
    }else{
      return false   
    }    
}
```

### 다시품
```jsx
function solution(x) {
    let num = String(x).split('').reduce((acc,cur) => acc+Number(cur), 0)
    if(x%num === 0) return true
    else return false
}
```
