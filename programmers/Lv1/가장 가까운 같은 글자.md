## 풀이
----
```jsx
function solution(s) {
 
    let arr=Array.from({length: s.length}, () => -1)
    for(let i=0; i<s.length; i++) {
        let count=0
        for(let j=i-1; j>=0; j--){
            count++
            if(s[i] === s[j]) {
                arr[i] = count
                break
            }
        }
    }
    return arr
}
```
