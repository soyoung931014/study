function solution(park, routes) {
    let result;
    const direction = {
        'E': [0,1],
        'S': [1,0],
        'W': [0,-1],
        'N': [-1,0]
    }
    routes= routes.map(el => el.split(" "))
  
    
    function findCoordinate(coorX,coorY){
        let mainX = coorX
        let mainY = coorY
        for(let route of routes){
            let dir = route[0]
            let cnt= Number(route[1])
            let flag=0
            let tmpX = mainX
            let tmpY = mainY
            for(let k=0; k<cnt; k++){
                tmpX += direction[dir][0]
                tmpY += direction[dir][1]
                if(tmpX >=0 && tmpY>=0 && tmpX<park.length && tmpY<park[0].length && (park[tmpX][tmpY] === 'O' || park[tmpX][tmpY] === 'S')){
                    continue
                   }else {
                       flag=1
                       break
                   }
            }
            if(!flag) {
                mainX = tmpX
                mainY = tmpY
            }
        }
        return [mainX,mainY]
    }
    
    for(let i=0; i<park.length; i++){
        for(let j=0; j<park[0].length; j++){
            if(park[i][j] === 'S') {
               result= findCoordinate(i,j)
               break
            }  
        }
    }
    return result
}
