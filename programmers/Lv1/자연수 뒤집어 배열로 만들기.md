-----

### 풀이
----
```jsx
function solution(n) {
   let result = String(n)
   let answer =[]
   for(let i = result.length-1; 0<=i; i--){
       answer.push(Number(result[i]))
   }
    return answer
}
```
### 또품
---   
```jsx
function solution(n) {
    let result = String(n).split("").map(el => Number(el)).reverse()
    return result
}
```
